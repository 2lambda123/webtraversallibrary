[docs]    @classmethod
    def wrapped(cls, func):
        """Wraps a function in a try-except block where AssertionErrors are logged and reraised as Error."""

        @wraps(func)
        def wrapper(*args, **kwargs):
            try:
                result = func(*args, **kwargs)
            except AssertionError as e:
                logger.error(e)
                raise cls(e)
            return result

        return wrapper
